20           List<Integer> sortedList = new ArrayList<>();

40           sortingThread0.setPriority(Thread.MAX_PRIORITY);

55           /* Runs the merge thread */

59       private static void merge(List<Integer> originalList, List<Integer> sortedList) { 
60           /* Variable for the half index of the list */ 
61           int halfIndex = originalList.size() / 2; 
62    
63           /* Runs until one half is totally consumed */ 
64           while (halfIndex > 0 && halfIndex < originalList.size()) { 
65               /* Checks the first elements of the halves */ 
66               if (originalList.get(0) <= originalList.get(halfIndex)) { 
67                   /* Adds the first element of the first half to the sorted list */ 
68                   sortedList.add(originalList.get(0)); 
69                   /* Removes added element */ 
70                   originalList.remove(0); 
71                   /* Decreases half index because of removal of the first element */ 
72                   halfIndex--; 
73               } else { 
74                   /* Adds the first element of the second half to the sorted list */ 
75                   sortedList.add(originalList.get(halfIndex)); 
76                   /* Removes added element */ 
77                   originalList.remove(halfIndex); 
78               } 
79           } 
80    
81           /* Adds rest of the elements of the original list to the sorted list */ 
82           sortedList.addAll(originalList); 
83    
84           /* Prints the sorted list */ 
85           System.out.println("Sorted List = " + sortedList); 
86       }
